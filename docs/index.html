Hadoop安装(单节点伪集群模式)

进入root账户

$ sudo su

解压hadoop到特定目录

$ tar xvzf hadoop-2.7.7.tar.gz -C /usr/local/

添加环境变量到.bashrc或者.zshrc

export HADOOP_HOME=/usr/local/hadoop-2.7.7
export HADOOP_CONF_DIR=$HADOOP_HOME/etc/hadoop
export HADOOP_MAPRED_HOME=$HADOOP_HOME
export HADOOP_COMMON_HOME=$HADOOP_HOME
export HADOOP_HDFS_HOME=$HADOOP_HOME
export YARN_HOME=$HADOOP_HOME
export HADOOP_COMMON_LIB_NATIVE_DIR=$HADOOP_HOME/lib/native
export HADOOP_OPTS="-Djava.library.path=$HADOOP_HOME/lib"

export PATH=$PATH:$HADOOP_HOME/bin

然后

$ source .bashrc
# or
$ source .zshrc

Hive安装

在root账户下

$ tar xvzf apache-hive-2.3.6-bin.tar.gz -C /usr/local

在.bashrc或者.zshrc中修改环境变量

$ export HIVE_HOME=/usr/local/apache-hive-2.3.6-bin
$ export PATH=$PATH:$HIVE_HOME/bin

然后

$ source .bashrc
# or
$ source .zshrc

然后运行以下命令

$ hdfs dfs -mkdir -p /user/hive/warehouse
$ hdfs dfs -mkdir /tmp

修改权限

$ hdfs dfs -chmod g+w /user/hive/warehouse
$ hdfs dfs -chmod g+w /tmp

注意我这里使用了MySQL 5.7来做Hive的元数据的管理。

在文件/usr/local/apache-hive-2.3.6-bin/conf/hive-env.sh中加入环境变量

$ export HADOOP_HOME=/usr/local/hadoop-2.7.7
$ export HADOOP_HEAPSIZE=512
$ export HIVE_CONF_DIR=/usr/local/apache-hive-2.3.6-bin/conf
$ export METASTORE_PORT=9083

初始化元数据存储

$ ./schematool -dbType mysql -initSchema root root

然后

$ hive --service metastore
$ hive

导入数据

腳本見sql腳本文件夾

$ mysql -u root -p
mysql> create database usertags charset=utf8;
mysql> use usertags;
mysql> source i_commodity.sql;
mysql> source i_marketing.sql;
mysql> source i_member.sql;
mysql> source i_operation.sql;
mysql> source i_order.sql;

解压sqoop压缩包

$ tar zxvf sqoop-1.4.6.bin__hadoop-2.0.4-alpha.tar.gz
$ mv sqoop-1.4.6.bin__hadoop-2.0.4-alpha sqoop

将jar包拷贝到sqoop中去

$ cp mysql-connector-java-5.1.28.jar ./sqoop/lib

重命名配置文件

$ cd /opt/sqoop/conf
$ mv sqoop-env-template.sh sqoop-env.sh

修改配置文件

$ vim sqoop-env.sh
$ export HIVE_HOME=/opt/hive

用jdbc连接mysql查看信息

$ ./sqoop list-databases --connect jdbc:mysql://localhost:3306/ --username root --password root

进入hive

$ hive

建库

hive>create database usertags;

导入数据脚本：

#!/bin/sh

sq()
{
./sqoop import \
--connect jdbc:mysql://localhost:3306/usertags \
--username root \
--password root \
--table $1 \
--num-mappers 1 \
--hive-import \
--fields-terminated-by "\t" \
--hive-overwrite \
--hive-database usertags \
--hive-table $1
}

sq t_commodity
sq t_commodity_cate
sq t_coupon
sq t_coupon_member
sq t_coupon_order
sq t_delivery
sq t_feedback
sq t_member
sq t_member_addr
sq t_order
sq t_order_commodity
sq t_shop
sq t_shop_order
sq t_user

将MySQL中的表都导入到Hive中。
